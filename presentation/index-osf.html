Bounce!

var WIDTH = 300;
var HEIGHT = 300;

function draw() {
    //clear screen and draw ball

    x += dx;
    y += dy;

    if (x > WIDTH || x < 0)
        dx = -dx;

    if (y > HEIGHT || y < 0)
        dy = -dy;
}

The Keyboard Part 1

function doKeyDown(evt) {
    if (evt.keyCode == 39)
        rightDown = true;
    else if (evt.keyCode == 37)
        leftDown = true;
}

function doKeyUp(evt) {
    if (evt.keyCode == 39)
        rightDown = false;
    else if (evt.keyCode == 37)
        leftDown = false;
}

$(document).keydown(doKeyDown);
$(document).keyup(doKeyUp);
