    ctx.clearRect(0,0,300,300);

    //draw the ball exactly as before

    x += dx;
    y += dy;
}

  • clear the screen every time

var WIDTH = 300;
var HEIGHT = 300;

function draw() {
    //clear screen and draw ball

    if (x > WIDTH || x < 0)
        dx = -dx;

    if (y > HEIGHT || y < 0)
        dy = -dy;
}
